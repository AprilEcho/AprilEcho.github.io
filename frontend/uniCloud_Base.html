# 创建方法

# 云函数

创建一个云函数

'use strict'
exports.main = async (event, context) => {
  //event为客户端上传的参数
  console.log('event : ', event)
  //返回数据给客户端
  return event
}

# 云对象

创建一个云对象

module.exports = {
  _before: function() {
    // 通用预处理器
  },
  async get(num) {
    //传递一个参数
    return await db
      .collection('article')
      .limit(num) //返回多少条数据
      .get()
  },
  add: async () => {
    await db.collection('article').add({
      title: '测试标题',
      content: '测试内容',
    })
  },
}

# Schema 学习

# 字段类型

"properties": {
  "_id": {
   "description": "ID，系统自动生成"
  },
  //自定义字段
  "title": {
    "bsonType": "string",//类型
    "title": "文章标题",//标题
    "description": "文章的标题",//描述
    "errorMessage": "标题必须填写"//错误描述,用于必填项
  },
  "posttime": {
    "bsonType": "timestamp",
    "title": "发布时间",
    "description": "发布的时间",
    "defaultValue": {
    "$env": "now"//时间戳
    }
  }
}
